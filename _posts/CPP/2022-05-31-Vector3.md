template<typename T>
class Iterator
{
public:
	Iterator() : _ptr(nullptr)
	{

	}

	{

	}

	Iterator& operator++()
	{
		_ptr++;
		return *this;
	}

	Iterator operator++(int)
	{
		Iterator temp = *this;
		_ptr++;
		return temp;
	}

	Iterator& operator--()
	{
		_ptr--;
		return *this;
	}

	{
		Iterator temp = *this;
		_ptr--;
		return temp;
	}

	{
		Iterator temp = *this;
		temp._ptr += count;
		return temp;
	}

	Iterator operator-(const int count)
	{
		Iterator temp = *this;
		temp._ptr -= count;
		return temp;

	{
		return _ptr == right._ptr;

	T* _ptr;
