## 🪐map


```cpp
#include <iostream>
using namespace std;
#include <map>

// map

class Player
{
public:
	Player() : _playerId(0) {}
	Player(int playerId) : _playerId(playerId) {}

public:
	int _playerId;
};

int main()
{
	// 연관 컨테이너

	// map : 균형 이진 트리 (AVL)
	// - 노드 기반

	class Node
	{
	public:
		Node* _left;
		Node* _right;
		// DATA
		pair<int, Player*> _data;
		//int _key;
		//Player* _value;
	};

	srand(static_cast<unsigned int>(time(nullptr)));
	// (Key, Value)
	map<int, int> m;

	// 10만명
	for (int i = 0; i < 100000; i++)
	{
		m.insert(pair<int, int>(i, i * 100));
	}

	// 5만명 퇴장
	for (int i = 0; i < 50000; i++)
	{
		int randomValue = rand() % 50000;

		// Erase By Key
		m.erase(randomValue);
	}

	// id가 1만 player를 빠르게 찾을 수 있다.

	map<int, int>::iterator findIt = m.find(100000);
	if (findIt != m.end())
	{
		cout << "Find!" << endl;
	}
	else
	{
		cout << "Can't find.." << endl;
	}

	return 0;
}
```
