#define _CRT_SECURE_NO_WARNINGS

#include <stdio.h>
#include <string.h>
#include <ctype.h>
#include<iostream>
#include <fstream>
#include <vector>
#include <algorithm>

using namespace std;

int n, k;
vector<int> v, dp;
void setting()
{
	cin >> n;
	v = vector<int>(n);
	for (int i = 0; i < n; ++i)
	{
		cin >> v[i];
	}

	cin >> k;
	dp = vector<int>(k + 1, 123456789);
	dp[0] = 0;
}

void solve()
{
	for (int i = 0; i < n; ++i)
	{
		int money = v[i];
		for (int j = money; j <= k; ++j)
		{
			dp[j] = min(dp[j], dp[j - money] + 1);
		}
	}

	cout << dp[k];
}

int main() 
{
	ios_base::sync_with_stdio(false);
	cin.tie(NULL);
	cout.tie(NULL);
	freopen("input.txt", "rt", stdin);

	setting();
	solve();

	return 0;
}
