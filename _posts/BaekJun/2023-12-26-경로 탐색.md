#define _CRT_SECURE_NO_WARNINGS

#include <stdio.h>
#include <string.h>
#include <ctype.h>
#include<iostream>
#include <fstream>
#include <vector>
#include <algorithm>

using namespace std;

int board[21][21];
int n, m, res = 0;
int visited[21];

void setting()
{
	cin >> n >> m;

	for (int i = 1; i <= m; ++i)
	{
		int a, b;
		cin >> a >> b;
		board[a][b] = 1;
	}
}

void DFS(int here)
{
	//cout << here << " ";
	if (here == n)
		res++;

	for (int there = 1; there <= n; ++there)
	{
		if (board[here][there] == 0)
			continue;

		if (visited[there] == 0)
		{
			visited[here] = 1;
			DFS(there);
			visited[here] = 0;
		}
	}
}

void solve()
{
	visited[1] = 1;
	DFS(1);

	cout << res;
}

int main()
{
	ios_base::sync_with_stdio(false);
	cin.tie(NULL);
	cout.tie(NULL);
	//freopen("input.txt", "rt", stdin);

	setting();
	solve();

	return 0;
}
