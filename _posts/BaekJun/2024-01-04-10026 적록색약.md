void BFS(int y, int x)
{
	if (discovered[y][x])
		return;
	char ch = board[y][x];
	if (ch == 'R')
		r++;
	else if (ch == 'G')
		g++;
	else
		b++;

	discovered[y][x] = 1;

	while (q.empty() == false)
	{
		pair<int, int> here = q.front();

			int nx = here.second + dx[i];

			if (ny < 1 || nx < 1 || ny > n || nx > n)
				continue;

				continue;
			if (board[ny][nx] != ch)

		}
	}
