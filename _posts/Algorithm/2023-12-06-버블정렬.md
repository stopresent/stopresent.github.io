```cpp
void bubbleSort(vector<int>& seq)
{
	for (int i = 0; i < seq.size(); ++i)
	{
		for (int j = 0; j < seq.size() - 1; ++j)
		{
			if (seq[j] > seq[j + 1])
				swap(seq[j], seq[j + 1]);
		}
	}
}
```
