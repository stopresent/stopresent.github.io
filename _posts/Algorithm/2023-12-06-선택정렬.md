```cpp
void selectionSort(vector<int>& seq)
{
	for (int i = 0; i < seq.size() - 1; ++i)
	{
		int idx = i;

		for (int j = i + 1; j < seq.size(); ++j)
		{
			if (seq[j] < seq[idx])
				idx = j;
		}

		swap(seq[i], seq[idx]);
	}
}
```
