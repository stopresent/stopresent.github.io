int n, key;
vector<int> seq;

void solve()
{
	cin >> n >> key;
	seq = vector<int>(n);
	for (int i = 0; i < n; ++i)
		cin >> seq[i];

	sort(seq.begin(), seq.end());

	int lt = 0, rt = n - 1, mid, res = 0;
	while (lt <= rt)
	{
		mid = (lt + rt) / 2;

		if (seq[mid] == key)
		{
			res = mid + 1;
			cout << res;
			return;
		}
		else if (seq[mid] > key)
		{
			rt = mid - 1;
		}
		else if (seq[mid] < key)
		{
			lt = mid + 1;
		}
	}
}
