```cpp
void insertionSort(vector<int>& seq)
{
	int cur;
	for (int i = 1; i < seq.size(); ++i)
	{
		cur = seq[i];
		for (int j = i - 1; j >= 0; --j)
		{
			if (cur < seq[j])
				swap(seq[j + 1], seq[j]);
		}
	}
}
```
